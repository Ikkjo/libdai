230template<class T>
231std::vector<T> concat( const std::vector<T>& u, const std::vector<T>& v ) {
232 std::vector<T> w;
233 w.reserve( u.size() + v.size() );
234 for( size_t i = 0; i < u.size(); i++ )
235 w.push_back( u[i] );
236 for( size_t i = 0; i < v.size(); i++ )
237 w.push_back( v[i] );
238 return w;
239}
